\begin{frame}[c]{Methodology}
  \begin{itemize}
  \item \structure{Declarativity} \
    The meaning of logic programs is independent of the order of its rules and their literals
  \item <2-> \structure{Question} \ What is a good logic program?
  \item \structure{Basic methodology}
    \begin{itemize}\normalsize
    \item Logic program \ = \ Data + Generator + Tester \ (\,+ Optimizer)
    \end{itemize}
  \item \structure{Refined methodology} \pause Easy Answer Set Programming (ezASP)
    \begin{itemize}\normalsize
    \item Logic program \ = \ List of rules ordered by predicate dependencies
    \end{itemize}
  \end{itemize}
\end{frame}
