\begin{frame}{Methodology}
  \bigskip
  \begin{itemize}
  \item \structure{Declarativity} \
    The meaning of logic programs is independent of
    \par
    the order of its rules and their literals
    \smallskip
  \item <2-> \structure{Question} \ What is a good logic program?
    \medskip
  \item <3-> \structure{Basic methodology}
    \begin{itemize}\normalsize
    \item Logic program \ = \ Data + Generator + Tester
    \end{itemize}
    \medskip
  \item <4-> \structure{Refined methodology} \pause[5] \ \alert{Easy Answer Set Programming} (ezASP)
    \begin{itemize}\normalsize
    \item<6-> Logic program \ = \ Rules ordered by predicate dependencies
      \medskip
    \item<7-> Construction of stable models
      \begin{itemize}\small
      \item successive applications of rules
      \item graphical representation
      \end{itemize}
    \end{itemize}
  \end{itemize}
\end{frame}
